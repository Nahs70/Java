import java.awt.Color;
import java.awt.Graphics;
import javax.swing.JFrame;
import javax.swing.JPanel;

public class ShapesDemo extends JPanel {

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g); // Clears background

        // Draw a blue line
        g.setColor(Color.BLUE);
        g.drawLine(50, 50, 200, 50);

        // Draw red rectangle outline
        g.setColor(Color.RED);
        g.drawRect(50, 100, 100, 80);

        // Fill green rectangle
        g.setColor(Color.GREEN);
        g.fillRect(200, 100, 100, 80);

        // Draw black oval outline
        g.setColor(Color.BLACK);
        g.drawOval(50, 220, 100, 60);

        // Fill orange circle
        g.setColor(Color.ORANGE);
        g.fillOval(200, 220, 60, 60);

        // Fill cyan rounded rectangle
        g.setColor(Color.CYAN);
        g.fillRoundRect(50, 320, 120, 60, 20, 20);
    }

    public static void main(String[] args) {
        JFrame frame = new JFrame("Basic Shapes with Graphics");
        ShapesDemo panel = new ShapesDemo();

        frame.add(panel); // FIXED (was 'panet')
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(400, 500);
        frame.setVisible(true);
    }
}
